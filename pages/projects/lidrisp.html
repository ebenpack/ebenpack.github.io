(define (map fn ls)
  (if (empty? ls) ls
    (cons (fn (car ls)) (map fn (cdr ls)))))
(define (double2 n) (+ n n))
(map double2 '(1 2 3 4 5))

(define (filter fn ls)
  (if (empty? ls)
    ls
    (if (fn (car ls))
          (cons (car ls) (filter fn (cdr ls)))
          (filter fn (cdr ls)))))
(define (even? n) (= 0 (modulo n 2)))
(filter even? '(1 2 3 4 5 6 7 8))

(define (fold fn acc ls)
  (if (empty? ls) acc
      (fold fn (fn acc (car ls)) (cdr ls))))
(fold + 0 '(1 2 3 4))
